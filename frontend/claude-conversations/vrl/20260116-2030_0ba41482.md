## Visual Effects for Hooks

### Color Flash Hook
```typescript
import { useCurrentFrame, interpolate, Easing } from "remotion";

export const ColorFlash: React.FC = () => {
  const frame = useCurrentFrame();
  const opacity = interpolate(
    frame,
    [0, 5, 10, 15],
    [0, 1, 0.3, 0],
    { extrapolateRight: "clamp" }
  );

  return (
    <div
      className="absolute inset-0 bg-red-500"
      style={{ opacity }}
    />
  );
};
```

### Screen Shake Effect
```typescript
export const ScreenShake: React.FC<{ children: React.ReactNode }> = ({ children }) => {
  const frame = useCurrentFrame();
  const shake = frame < 30 ? Math.sin(frame * 2) * 5 : 0;

  return (
    <div style={{ transform: `translate(${shake}px, ${shake * 0.5}px)` }}>
      {children}
    </div>
  );
};
```
